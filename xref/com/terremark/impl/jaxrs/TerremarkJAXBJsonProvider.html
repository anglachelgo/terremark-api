51  /**
52   * Custom JAXB JSON provider to deal with Terremark specific content type and custom deserialization.
53   * <p/>
54   * NOTE: This is currently work-in-progress. JSON content-type does not currently work.
55   *
56   * @author <a href="mailto:spasam@terremark.com">Seshu Pasam</a>
57   */
58  @Provider
59  @Produces({TerremarkConstants.TERREMARK_MEDIA_TYPE_WILDCARD})
60  @Consumes({TerremarkConstants.TERREMARK_MEDIA_TYPE_WILDCARD})
61  public final class TerremarkJAXBJsonProvider extends JacksonJsonProvider {
62      /** Deserializer version */
63      private static final Version DESERIALIZERS_VERSION = new Version(1, 0, 0, null);
64  
65      /**
66       * Default constructor. Registers the deserializer modules.
67       */
68      public TerremarkJAXBJsonProvider() {
69          super(createObjectMapper(), new Annotations[] {Annotations.JAXB});
70      }
71  
72      /**
73       * Terremark content type uses special "application/vnd.tmrk..." format to indicate the type of response object.
74       * This does not work well with the Jackson JSON provider. So override this method and make sure that content can be
75       * read.
76       *
77       * @see org.codehaus.jackson.jaxrs.JacksonJsonProvider#isReadable(java.lang.Class, java.lang.reflect.Type,
78       * java.lang.annotation.Annotation[], javax.ws.rs.core.MediaType)
79       */
80      @Override
81      public boolean isReadable(final Class<?> type, final Type genericType, final Annotation[] annotations,
82                      final MediaType mediaType) {
83          return super.isReadable(type, genericType, annotations, mediaType)
84                          || mediaType.getSubtype().startsWith(TerremarkConstants.TERREMARK_SUB_TYPE_PREFIX);
85      }
86  
87      /**
88       * Configures the mapper and adds the custom deserialization modules.
89       *
90       * @return Object mapper.
91       */
92      private static ObjectMapper createObjectMapper() {
93          final ObjectMapper mapper = new ObjectMapper();
94  
95          final SimpleModule svModule = new SimpleModule("SupportedVersionsDeserializer", DESERIALIZERS_VERSION);
96          svModule.addDeserializer(SupportedVersions.class, new SupportedVersionsDeserializer());
97          mapper.registerModule(svModule);
98  
99          final SimpleModule dateModule = new SimpleModule("XMLGregorianCalendarDeserializer", DESERIALIZERS_VERSION);
100         dateModule.addDeserializer(XMLGregorianCalendar.class, new XMLGregorianCalendarDeserializer());
101         mapper.registerModule(dateModule);
102 
103         mapper.configure(Feature.FAIL_ON_UNKNOWN_PROPERTIES, true);
104         mapper.setDateFormat(StdDateFormat.getBlueprintISO8601Format());
105         mapper.setAnnotationIntrospector(new JaxbAnnotationIntrospector());
106 
107         return mapper;
108     }
109 
110     /**
111      * Custom JSON deserializer for {@code SupportedVersions}.
112      *
113      * @author <a href="mailto:spasam@terremark.com">Seshu Pasam</a>
114      */
115     static class SupportedVersionsDeserializer extends JsonDeserializer<SupportedVersions> {
116         /**
117          * (non-Javadoc)
118          *
119          * @see org.codehaus.jackson.map.JsonDeserializer#deserialize(org.codehaus.jackson.JsonParser,
120          * org.codehaus.jackson.map.DeserializationContext)
121          */
122         @Override
123         public SupportedVersions deserialize(final JsonParser jp, final DeserializationContext ctxt)
124                         throws IOException, JsonProcessingException {
125             final List<VersionInfo> versions = ctxt.getParser().readValueAs(new TypeReference<List<VersionInfo>>() {
126                 // There is nothing to implement. This anonymous class is used just to pass the Java Generic
127                 // information to the reader. In this case, the generic information is List<VersionInfo>
128             });
129 
130             final SupportedVersions sv = new SupportedVersions();
131             sv.getVersionInfos().addAll(versions);
132 
133             return sv;
134         }
135     }

137     /**
138      * Custom JSON deserializer for {@code XMLGregorianCalendar}.
139      *
140      * @author <a href="mailto:spasam@terremark.com">Seshu Pasam</a>
141      */
142     static class XMLGregorianCalendarDeserializer extends JsonDeserializer<XMLGregorianCalendar> {
143         /** JSON date prefix */
144         private static final String JSON_DATE_PREFIX = "/Date(";
145         /** Date prefix length */
146         private static final int JSON_DATE_PREFIX_LENGTH = JSON_DATE_PREFIX.length();
147         /** JSON date suffix */
148         private static final String JSON_DATE_SUFFIX = ")/";
149         /** Date suffix length */
150         private static final int JSON_DATE_SUFFIX_LENGTH = JSON_DATE_SUFFIX.length();
151         /** Cached data type factory for creating gregorian calendar */
152         private static final DatatypeFactory DATA_TYPE_FACTORY;
153 
154         static {
155             try {
156                 DATA_TYPE_FACTORY = DatatypeFactory.newInstance();
157             } catch (final DatatypeConfigurationException ex) {
158                 throw new IllegalArgumentException(ex);
159             }
160         }
161 
162         /**
163          * (non-Javadoc)
164          *
165          * @see org.codehaus.jackson.map.JsonDeserializer#deserialize(org.codehaus.jackson.JsonParser,
166          * org.codehaus.jackson.map.DeserializationContext)
167          */
168         @Override
169         public XMLGregorianCalendar deserialize(final JsonParser jp, final DeserializationContext ctxt)
170                         throws IOException, JsonProcessingException {
171             if (jp.getCurrentToken() == JsonToken.VALUE_STRING) {
172                 String text = jp.getText().trim();
173 
174                 if (text.startsWith(JSON_DATE_PREFIX) && text.endsWith(JSON_DATE_SUFFIX)) {
175                     text = text.substring(JSON_DATE_PREFIX_LENGTH, text.length() - JSON_DATE_SUFFIX_LENGTH);
176 
177                     final GregorianCalendar gc = new GregorianCalendar();
178                     gc.setTimeInMillis(Long.parseLong(text));
179 
180                     return DATA_TYPE_FACTORY.newXMLGregorianCalendar(gc);
181                 }
182             }
183 
184             throw ctxt.mappingException(XMLGregorianCalendar.class, jp.getCurrentToken());
185         }
186     }
