18  import java.util.Map;

20  import com.terremark.exception.TerremarkException;

22  /**
23   * Interface that can be implemented to specify custom retry strategy. It is possible that API calls can fail randomly.
24   * When a {@code HTTP GET} call fails with error code greater than 499, the call can be retried. User's can determine if
25   * a specific failed call needs to be retried and how many times. Other types of HTTP request (such as POST, PUT,
26   * DELETE) cannot be retried. Implementation of this interface will only be invoked for HTTP GET calls that fail with
27   * 500 or greater error code.
28   *
29   * @author <a href="mailto:spasam@terremark.com">Seshu Pasam</a>
30   * @see com.terremark.config.PropertiesBuilder#setRetryHandler(RetryHandler)
31   */
32  public interface RetryHandler {
33      /**
34       * Method invoked to determine if the API call should be retried or not.
35       *
36       * @param failuresSoFar Number of failures so far. Starts with {@code 1} for the first failure.
37       * @param exception Terremark exception. This may contain information like HTTP error code, message etc.
38       * @param relativePath Relative URL path that was being accessed for the API call.
39       * @param queryParams Query arguments being passed to the API end point.
40       * @param extraHeaders Additional headers passed to the API end point.
41       * @param responseClass Type of response excepted from the API end point.
42       * @param arguments Additional arguments being passed to the end point.
43       * @return Implementation should return {@code true} to indicate that the call should be retried. If {@code false}
44       * is returned, the call is not retried and {@link TerremarkException} will be thrown.
45       */
46      <T> boolean shouldRetry(int failuresSoFar, TerremarkException exception, final String relativePath,
47                      final Map<String, String> queryParams, final Map<String, String> extraHeaders,
48                      final Class<T> responseClass, final Object... arguments);
49  }
