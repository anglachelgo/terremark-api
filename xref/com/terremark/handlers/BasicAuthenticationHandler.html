16  package com.terremark.handlers;
17  
18  import java.io.UnsupportedEncodingException;
19  
20  import javax.ws.rs.core.HttpHeaders;
21  
22  import org.apache.commons.codec.binary.Base64;
23  import org.apache.wink.client.ClientRequest;
24  import org.apache.wink.client.ClientResponse;
25  import org.apache.wink.client.handlers.ClientHandler;
26  import org.apache.wink.client.handlers.HandlerContext;
27  import org.apache.wink.client.internal.handlers.ClientRequestImpl;
28  
29  import com.terremark.annotations.AuthenticationNotRequired;
30  
31  /**
32   * Basic authentication handler. If the client is configured for basic authentication this handler should be added to
33   * the chain. This handler will add the {@code Authorization} header before the request is sent to the server.
34   *
35   * @author <a href="mailto:spasam@terremark.com">Seshu Pasam</a>
36   */
37  public class BasicAuthenticationHandler implements ClientHandler {
38      /** Authorization header value, computed once and saved */
39      private transient final String authHeader;
40  
41      /**
42       * Default constructor. Takes the credentials and pre-computes the {@code Authorization} header.
43       *
44       * @param userName User name.
45       * @param password Password.
46       */
47      public BasicAuthenticationHandler(final String userName, final String password) {
48          try {
49              final String credentials = userName + ":" + password;
50              final byte[] encodedCredentials = Base64.encodeBase64(credentials.getBytes("UTF-8"), false);
51              this.authHeader = "Basic " + new String(encodedCredentials, "UTF-8");
52          } catch (UnsupportedEncodingException ex) {
53              throw new IllegalArgumentException(ex);
54          }
55      }
56  
57      /**
58       * Method invoked in the chain. If the request entity class is not annotated with
59       * {@link com.terremark.annotations.AuthenticationNotRequired AuthenticationNotRequired} this method will add the
60       * {@code Authorization} header.
61       *
62       * @param request Client request.
63       * @param context Request context.
64       * @return Client response.
65       * @throws Exception If there is a problem invoking the chain.
66       */
67      @Override
68      @SuppressWarnings("PMD.SignatureDeclareThrowsException")
69      public final ClientResponse handle(final ClientRequest request, final HandlerContext context) throws Exception {
70          final Class<?> responseEntityClass = (Class<?>) request.getAttributes().get(
71                          ClientRequestImpl.RESPONSE_ENTITY_CLASS_TYPE);
72          if (!responseEntityClass.isAnnotationPresent(AuthenticationNotRequired.class)) {
73              request.getHeaders().putSingle(HttpHeaders.AUTHORIZATION, authHeader);
74          }
75  
76          return context.doChain(request);
77      }
78  }
