142         DeviceLayout layout = client.getEnvironmentHandler().getLayouts(environmentId);

145         // Create row 1
146         CreateLayoutRow createLayoutRow = new CreateLayoutRow();
147         createLayoutRow.setName("Row 1" + TEST_ID);
148         LayoutRow row1 = client.getEnvironmentHandler().layoutRowCreate(environmentId, createLayoutRow);
149         assertResource(row1);
150 
151         // Create row 2
152         createLayoutRow.setName("Row 2" + TEST_ID);
153         LayoutRow row2 = client.getEnvironmentHandler().layoutRowCreate(environmentId, createLayoutRow);
154         assertResource(row2);
155 
156         // Create group 1 in row 1
157         CreateLayoutGroup createLayoutGroup = new CreateLayoutGroup();
158         createLayoutGroup.setRow(TerremarkFactory.getReference(row1));
159         createLayoutGroup.setName("Group 1" + TEST_ID);
160         LayoutGroup group11 = client.getEnvironmentHandler().layoutGroupCreate(environmentId, createLayoutGroup);
161         assertResource(group11);
162 
163         // Create group 2 in row 1
164         createLayoutGroup.setName("Group 2" + TEST_ID);
165         LayoutGroup group12 = client.getEnvironmentHandler().layoutGroupCreate(environmentId, createLayoutGroup);
166         assertResource(group12);
167 
168         // Get row 1 and group 1 by ID
169         assertResource(client.getEnvironmentHandler().getLayoutRowByID(TerremarkFactory.getId(row1)));
170         assertResource(client.getEnvironmentHandler().getLayoutGroupByID(TerremarkFactory.getId(group11)));
171 
172         // Edit row 1
173         row1.setName("New Row 1" + TEST_ID);
174         client.getEnvironmentHandler().layoutRowEdit(row1);
175 
176         // Edit group 1
177         group11.setName("New Group 1" + TEST_ID);
178         client.getEnvironmentHandler().layoutGroupEdit(group11);
179 
180         // Move row 1 down and then up
181         client.getEnvironmentHandler().layoutRowMoveDown(TerremarkFactory.getId(row1));
182         client.getEnvironmentHandler().layoutRowMoveUp(TerremarkFactory.getId(row1));
183 
184         // Move group 1 down and then up. The following calls fail in Livespec environment
185         //client.getEnvironmentHandler().layoutGroupOrderLower(TerremarkFactory.getId(group11));
186         //client.getEnvironmentHandler().layoutGroupOrderHigher(TerremarkFactory.getId(group11));
187 
188         // Remove all groups and rows
189         client.getEnvironmentHandler().layoutGroupRemove(TerremarkFactory.getId(group11));
190         client.getEnvironmentHandler().layoutGroupRemove(TerremarkFactory.getId(group12));
191         client.getEnvironmentHandler().layoutRowRemove(TerremarkFactory.getId(row1));
192         client.getEnvironmentHandler().layoutRowRemove(TerremarkFactory.getId(row2));
